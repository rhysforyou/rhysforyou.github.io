The Coding Process

First up we write the file header in which we include any external code we plan to use and set the namespace.

#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	double x, y, z;
	double result;
	    
	cout << "Enter a number x = ";
	cin >> x;

	cout << "Enter a number y = ";
	cin >> y;

	cout << "Enter a number z = ";
	cin >> z;

	result = sqrt(x - y) /(z - x * y);

	cout << "sqrt(x - y) /(z - x * y) = " << result << endl;
	return 0;
}
